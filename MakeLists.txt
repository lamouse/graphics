cmake_minimum_required(VERSION 3.19)
project(graphics)

set(EXECUTE_NAME graphics)

set(CMAKE_CXX_STANDARD 23)

set(librariesList
    GLFW
    vulkan
)

find_library(${librariesList})
include_directories("/usr/local/include")
link_directories("/usr/local/lib")

add_executable(${EXECUTE_NAME} 
        src/main.cpp 
        src/g_app.cpp 
        src/g_window.cpp 
        src/g_pipline.cpp 
        src/g_context.cpp 
        src/g_device.cpp
)

target_link_libraries(${EXECUTE_NAME} ${librariesList})


set(CPACK_PACKAGE_DESCRIPTION_FILE "${PROJECT_SOURCE_DIR}/Readme.md")
set(CPACK_GENERATOR "ZIP;TGZ")
file(GLOB SHADER_FILE
        "shader/*.spv"
        )
file(COPY ${SHADER_FILE} DESTINATION shader)
install(DIRECTORY "shader/"
        DESTINATION "shader"
        FILES_MATCHING PATTERN "*.spv" )

set(CMAKE_VERBOSE_MAKEFILE ON)
